#include "bsp/bsp.h"
#include "console/console.h"
#include "hal/hal_gpio.h"
#include "app_gpio.h"
#include "collector.h"
#include "hal_lptimer.h"

#define TEST_LPTIMER

int button_device[2] = {
	USER_BLUE_BUTTON,
};




static void gpio_irq_handler(void *arg)
{

}

void app_gpio_init(void)
{
	/* Buttons configiuratin & setting */
	hal_gpio_irq_init(button_device[0], gpio_irq_handler, NULL,
			  HAL_GPIO_TRIG_RISING, HAL_GPIO_PULL_NONE);

    if(1){

		hal_gpio_init_out(MCU_GPIO_PORTC(9), 0);
		
		hal_gpio_irq_enable(button_device[0]);

#ifdef TEST_LPTIMER
		hal_lptimer_init();
		hal_lptimer_start(10000);
#endif		

	}

	start_datacollector();

}
